from django.shortcuts import render
from PyPDF2 import PdfReader
import docx

def extract_text(file):
    ext = file.name.split(".")[-1].lower()

    if ext == "pdf":
        reader = PdfReader(file)
        return "\n".join(page.extract_text() or "" for page in reader.pages)

    elif ext == "docx":
        doc = docx.Document(file)
        return "\n".join(p.text for p in doc.paragraphs)

    elif ext == "txt":
        return file.read().decode("utf-8")

    return ""

@login_required
def dashboard(request):
 
    trips = Trip.objects.filter(user=request.user)
 
    flights = Flight.objects.filter(
        trip__user=request.user
    )
 
    hotels = HotelBooking.objects.filter(
        trip__user=request.user
    )
 
    events = CalendarEvent.objects.filter(
        trip__user=request.user
    )
 
    upcoming_trips = trips.filter(
        start_date__gte=timezone.now().date()
    )
 
    context = {
        "trip_count": trips.count(),
        "flight_count": flights.count(),
        "hotel_count": hotels.count(),
        "event_count": events.count(),
        "upcoming_trips": upcoming_trips[:5],
    }
 
    return render(
        request,
        "dashboard.html",
        context
    )

from django.http import JsonResponse
 
@login_required
def calendar_events(request):
 
    events = CalendarEvent.objects.filter(
        trip__user=request.user
    )
 
    data = []
 
    for event in events:
 
        data.append({
            "title": event.title,
            "start": event.start_datetime.isoformat(),
            "end": event.end_datetime.isoformat(),
            "color": event.color_code,
        })
 
    return JsonResponse(
        data,
        safe=False
    )

@login_required
def export_event_ics(request, event_id):

UID:{event.id}@travelplanner.com

        content,
        content_type="text/calendar"
    )
 
    response["Content-Disposition"] = (

    )
 
    return response

@login_required
def export_trip_ics(request, trip_id):
 
    trip = get_object_or_404(
        Trip,
        id=trip_id,
        user=request.user
    )
 
    ics = """
BEGIN:VCALENDAR
VERSION:2.0
PRODID:-//Travel Planner//EN
"""
 
    # Flights
    for flight in trip.flights.all():
 
        ics += f"""
BEGIN:VEVENT
UID:flight-{flight.id}
SUMMARY:Flight {flight.airline} {flight.flight_number}
DTSTART:{flight.departure_datetime.strftime('%Y%m%dT%H%M%S')}
DTEND:{flight.arrival_datetime.strftime('%Y%m%dT%H%M%S')}
LOCATION:{flight.departure_airport} → {flight.arrival_airport}
DESCRIPTION:Booking Ref {flight.booking_reference}
END:VEVENT
"""
 
    # Hotels
    for hotel in trip.hotels.all():
 
        ics += f"""
BEGIN:VEVENT
UID:hotel-{hotel.id}
SUMMARY:Hotel - {hotel.hotel_name}
DTSTART:{hotel.checkin_date.strftime('%Y%m%d')}
DTEND:{hotel.checkout_date.strftime('%Y%m%d')}
LOCATION:{hotel.location}
DESCRIPTION:Booking Ref {hotel.booking_reference}
END:VEVENT
"""
 
    # Manual Events
    for event in trip.events.all():
 
        ics += f"""
BEGIN:VEVENT
UID:event-{event.id}
SUMMARY:{event.title}
DTSTART:{event.start_datetime.strftime('%Y%m%dT%H%M%S')}
DTEND:{event.end_datetime.strftime('%Y%m%dT%H%M%S')}
END:VEVENT
"""
 
    ics += "\nEND:VCALENDAR"
 
    response = HttpResponse(
        ics,
        content_type="text/calendar"
    )
 
    filename = trip.trip_name.replace(" ", "_")
 
    response[
        "Content-Disposition"
    ] = f'attachment; filename="{filename}.ics"'
 
    return response

class Notification(models.Model):
 
    NOTIFICATION_TYPES = [
        ("trip", "Trip"),
        ("flight", "Flight"),
        ("hotel", "Hotel"),
        ("event", "Event"),
    ]
 
    user = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name="notifications"
    )
 
    notification_type = models.CharField(
        max_length=20,
        choices=NOTIFICATION_TYPES
    )
 
    title = models.CharField(
        max_length=255
    )
 
    message = models.TextField()
 
    is_read = models.BooleanField(
        default=False
    )
 
    created_at = models.DateTimeField(
        auto_now_add=True
    )
 
    def __str__(self):
        return self.title

Step 4: Mark Notification Read
View



Python
@login_required
def mark_notification_read(
    request,
    notification_id
):
 
    notification = get_object_or_404(
        Notification,
        id=notification_id,
        user=request.user
    )
 
    notification.is_read = True
 
    notification.save()
 
    return redirect(
        "homepage:dashboard"
    )

Step 8: Notification Center Page
Create:




Python
@login_required
def notification_list(
    request
):
 
    notifications = Notification.objects.filter(
        user=request.user
    ).order_by("-created_at")
 
    return render(
        request,
        "notifications.html",
        {
            "notifications": notifications
        }
    )
